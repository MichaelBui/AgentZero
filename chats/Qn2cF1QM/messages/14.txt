103 def _call_llm(prompt: str) -> str:
104     """Call LiteLLM proxy for summarization."""
105     url = f"{LITELLM_BASE}/chat/completions"
106     headers = {"Content-Type": "application/json"}
107     if LITELLM_API_KEY:
108         headers["Authorization"] = f"Bearer {LITELLM_API_KEY}"
109 
110     payload = {
111         "model": SUMMARIZE_MODEL,
112         "messages": [{"role": "user", "content": prompt}],
113         "temperature": 0.3,
114     }
115 
116     try:
117         resp = requests.post(url, headers=headers, json=payload, timeout=120)
118         if resp.status_code != 200:
119             eprint(f"WARN: Summarization API returned {resp.status_code}: {resp.text[:200]}")
120             return ""
121         data = resp.json()
122         content = data.get("choices", [{}])[0].get("message", {}).get("content", "")
123         if content.startswith("<think>"):
124             think_end = content.find("</think>")
125             if think_end != -1:
126                 content = content[think_end + len("</think>"):].strip()
127         return content.strip()
128     except requests.exceptions.Timeout:
129         eprint("WARN: Summarization request timed out (120s)")
130         return ""
131     except Exception as e:
132         eprint(f"WARN: Summarization failed: {e}")
133         return ""
