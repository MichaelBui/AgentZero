 25 def summarize_resource(
 26     title: str,
 27     source_type: str,
 28     atomic_items: list[dict],
 29     metadata: Optional[dict] = None,
 30     existing_summary: Optional[str] = None,
 31 ) -> str:
 32     """Generate a summary for an email thread from its atomic messages.
 33 
 34     If existing_summary is provided, uses an anchored update approach
 35     to extend the existing summary with new content, preserving historical context.
 36     """
 37     if not atomic_items:
 38         return existing_summary or ""
 39 
 40     content_parts = []
 41     for item in atomic_items:
 42         author = item.get("author", "Unknown")
 43         ts = item.get("created_at", "")
 44         text = item.get("content", "")
 45         if text:
 46             content_parts.append(f"[{ts}] {author}: {text}")
 47 
 48     content_block = "\n---\n".join(content_parts)
 49     meta_json = json.dumps(metadata or {}, indent=2, ensure_ascii=False)
 50 
 51     if existing_summary:
 52         prompt = _build_update_prompt(title, source_type, meta_json, existing_summary, content_block)
 53     else:
 54         prompt = _build_full_prompt(title, source_type, meta_json, content_block)
 55 
 56     return _call_llm(prompt)
