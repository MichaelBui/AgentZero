 52 def get_current_branch(path, dry_run=False):
 53     """Get the current branch name."""
 54     result = run_command(
 55         ["git", "branch", "--show-current"],
 56         cwd=path,
 57         dry_run=dry_run,
 58         check=False
 59     )
 60     branch = result.stdout.strip()
 61     if not branch:
 62         # Detached HEAD or other issue, try to get from ref
 63         result = run_command(
 64             ["git", "rev-parse", "--abbrev-ref", "HEAD"],
 65             cwd=path,
 66             dry_run=dry_run,
 67             check=False
 68         )
 69         branch = result.stdout.strip()
 70     return branch or "main"

107 def is_ahead_of_remote(path, branch, dry_run=False):
108     """Check if local is ahead of remote."""
109     result = run_command(
110         ["git", "rev-list", "--count", f"origin/{branch}..HEAD"],
111         cwd=path,
112         dry_run=dry_run,
113         check=False
114     )
115     try:
116         count = int(result.stdout.strip())
117         return count > 0
118     except (ValueError, AttributeError):
119         return False
