 14     sys.exit(1)

 16 # ntfy.sh endpoint
 17 NTFY_URL = "https://ntfy.sh/AgentZero-eHqWuKJQWSuKmXBE"
 18 
 19 def eprint(*args, **kwargs):
 20     print(*args, file=sys.stderr, **kwargs)
 21 
 22 
 23 def send_notification(message: str, title: str = "Agent Zero Notification", priority: int = 3) -> dict:
 24     """Send a notification via ntfy.sh. Returns result dict."""
 25     result = {"success": False, "title": title, "priority": priority, "message": message}
 26     try:
 27         headers = {
 28             "Title": title,
 29             "Priority": str(priority)
 30         }
 31         
 32         response = requests.post(
 33             NTFY_URL,
 34             data=message.encode('utf-8'),
 35             headers=headers,
 36             timeout=10
 37         )
 38         
 39         if response.status_code == 200:
 40             eprint("Notification sent successfully")
 41             result["success"] = True
 42         else:
 43             result["error"] = f"HTTP {response.status_code}: {response.text}"
 44             eprint(f"Failed to send notification: {result['error']}")
 45             
 46     except requests.exceptions.Timeout:
 47         result["error"] = "Request timed out"
 48         eprint(result["error"])
 49     except requests.exceptions.ConnectionError:
 50         result["error"] = "Could not connect to ntfy.sh"
 51         eprint(result["error"])
 52     except Exception as e:
 53         result["error"] = str(e)
 54         eprint(f"Error: {e}")
 55 
 56     return result
 57 
 58 
 59 def output_data(data, fmt: str = "json"):
 60     """Output data as JSON (default) or YAML to stdout."""
 61     if fmt == "yaml":
 62         import yaml
 63         print(yaml.dump(data, default_flow_style=False, allow_unicode=True, sort_keys=False))
 64     else:
 65         print(json.dumps(data, separators=(",", ":"), ensure_ascii=False))
