 25 def run_command(cmd, cwd, dry_run=False, check=True):
 26     """Run a shell command and return the result."""
 27     if dry_run:
 28         print(f"  [DRY-RUN] Would run: {' '.join(cmd)}")
 29         return subprocess.CompletedProcess(cmd, 0, b"", b"")
 30     
 31     result = subprocess.run(
 32         cmd,
 33         cwd=cwd,
 34         capture_output=True,
 35         text=True
 36     )
 37     
 38     if check and result.returncode != 0:
 39         raise subprocess.CalledProcessError(
 40             result.returncode, cmd, result.stdout, result.stderr
 41         )
 42     
 43     return result

 52 def get_current_branch(path, dry_run=False):
 53     """Get the current branch name."""
 54     result = run_command(
 55         ["git", "branch", "--show-current"],
 56         cwd=path,
 57         dry_run=dry_run,
 58         check=False
 59     )
 60     branch = result.stdout.strip()
 61     if not branch:
 62         # Detached HEAD or other issue, try to get from ref
 63         result = run_command(
 64             ["git", "rev-parse", "--abbrev-ref", "HEAD"],
 65             cwd=path,
 66             dry_run=dry_run,
 67             check=False
 68         )
 69         branch = result.stdout.strip()
 70     return branch or "main"

 73 def has_remote(path, dry_run=False):
 74     """Check if the repository has a remote origin configured."""
 75     result = run_command(
 76         ["git", "remote", "get-url", "origin"],
 77         cwd=path,
 78         dry_run=dry_run,
 79         check=False
 80     )
 81     return result.returncode == 0

 84 def has_staged_changes(path, dry_run=False):
 85     """Check if there are staged changes ready to commit."""
 86     result = run_command(
 87         ["git", "diff", "--cached", "--quiet"],
 88         cwd=path,
 89         dry_run=dry_run,
 90         check=False
 91     )
 92     # Returns 0 if no changes, 1 if there are changes
 93     return result.returncode != 0

 96 def has_any_changes(path, dry_run=False):
 97     """Check if there are any changes (staged or unstaged)."""
 98     result = run_command(
 99         ["git", "status", "--porcelain"],
100         cwd=path,
101         dry_run=dry_run,
102         check=False
103     )
104     return bool(result.stdout.strip())

107 def is_ahead_of_remote(path, branch, dry_run=False):
108     """Check if local is ahead of remote."""
109     result = run_command(
110         ["git", "rev-list", "--count", f"origin/{branch}..HEAD"],
111         cwd=path,
112         dry_run=dry_run,
113         check=False
114     )
115     try:
116         count = int(result.stdout.strip())
117         return count > 0
118     except (ValueError, AttributeError):
119         return False

122 def backup_path(path, custom_message=None, dry_run=False):
123     """
124     Backup a single path to its git remote.
125     
126     Git Strategy:
127     1. Stage all changes (git add -A)
128     2. Commit locally with timestamp (if there are changes)
129     3. Fetch remote
130     4. Merge with -X ours (local changes priority, remote preserved in history)
131     5. Push to remote (if there's something to push)
132     """
133     print(f"\n{'='*60}")
134     print(f"Backing up: {path}")
135     print(f"{'='*60}")
136     
137     path_obj = Path(path)
138     
139     # Validate path exists
140     if not path_obj.exists():
141         print(f"  ❌ Error: Path does not exist: {path}")
142         return False
143     
144     # Validate git repository
145     if not is_git_repo(path):
146         print(f"  ❌ Error: Not a git repository: {path}")
147         print(f"     Run: git init && git remote add origin <url>")
148         return False
149     
150     # Validate remote exists
151     if not has_remote(path, dry_run):
152         print(f"  ❌ Error: No remote origin configured for: {path}")
153         print(f"     Run: git remote add origin <url>")
154         return False
155     
156     branch = get_current_branch(path, dry_run)
157     print(f"  📋 Branch: {branch}")
158     
159     # Check for any changes
160     has_changes = has_any_changes(path, dry_run)
161     
162     if not has_changes:
163         print(f"  ℹ️  No local changes detected")
164         
165         # Still try to fetch and sync with remote
166         print(f"  🔄 Fetching remote...")
167         try:
168             run_command(["git", "fetch", "origin"], cwd=path, dry_run=dry_run, check=False)
169         except subprocess.CalledProcessError as e:
170             print(f"  ⚠️  Warning: Could not fetch remote: {e.stderr}")
171         
172         # Check if we need to pull (local behind remote)
173         result = run_command(
174             ["git", "rev-list", "--count", f"HEAD..origin/{branch}"],
175             cwd=path,
176             dry_run=dry_run,
177             check=False
178         )
179         
180         try:
181             behind_count = int(result.stdout.strip()) if result.stdout.strip() else 0
182         except ValueError:
183             behind_count = 0
184         
185         if behind_count > 0:
186             print(f"  📥 Local is behind remote by {behind_count} commits, pulling...")
187             try:
188                 run_command(["git", "pull", "origin", branch], cwd=path, dry_run=dry_run, check=False)
189                 print(f"  ✅ Synced with remote")
190             except subprocess.CalledProcessError as e:
191                 print(f"  ⚠️  Warning: Could not pull: {e.stderr}")
192         else:
193             print(f"  ✅ Already in sync with remote - nothing to backup")
194         
195         return True
196     
197     # Step 1: Stage all changes
198     print(f"  📦 Staging changes...")
199     try:
200         run_command(["git", "add", "-A"], cwd=path, dry_run=dry_run)
201     except subprocess.CalledProcessError as e:
202         print(f"  ❌ Failed to stage changes: {e.stderr}")
203         return False
204     
205     # Check if staging resulted in anything to commit
206     if not has_staged_changes(path, dry_run) and not dry_run:
207         print(f"  ℹ️  No changes to commit after staging")
208         return True
209     
210     # Step 2: Commit locally
211     timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
212     if custom_message:
213         commit_message = f"{custom_message} [{timestamp}]"
214     else:
215         commit_message = f"Backup [{timestamp}]"
216     
217     print(f"  💾 Committing: {commit_message}")
218     result = run_command(
219         ["git", "commit", "-m", commit_message],
220         cwd=path,
221         dry_run=dry_run,
222         check=False
223     )
224     
225     if result.returncode != 0:
226         # Check if it's just "nothing to commit"
227         if "nothing to commit" in result.stdout.lower() or "nothing to commit" in result.stderr.lower():
228             print(f"  ℹ️  Nothing to commit - working tree clean")
229         else:
230             print(f"  ⚠️  Commit warning: {result.stderr}")
231             # Continue anyway, might still have unpushed commits
232     
233     # Step 3: Fetch remote
234     print(f"  🔄 Fetching remote...")
235     try:
236         run_command(["git", "fetch", "origin"], cwd=path, dry_run=dry_run, check=False)
237     except subprocess.CalledProcessError as e:
238         print(f"  ⚠️  Warning: Could not fetch remote: {e.stderr}")
239     
240     # Step 4: Merge with local priority (-X ours)
241     print(f"  🔀 Checking remote state...")
242     
243     # Check if remote branch exists
244     result = run_command(
245         ["git", "rev-parse", f"origin/{branch}"],
246         cwd=path,
247         dry_run=dry_run,
248         check=False
249     )
250     
251     remote_exists = result.returncode == 0
252     
253     if remote_exists:
254         # Check if we need to merge (remote has commits we don't have)
255         result = run_command(
256             ["git", "rev-list", "--count", f"HEAD..origin/{branch}"],
257             cwd=path,
258             dry_run=dry_run,
259             check=False
260         )
261         
262         try:
263             commits_behind = int(result.stdout.strip()) if result.stdout.strip() else 0
264         except ValueError:
265             commits_behind = 0
266         
267         if commits_behind > 0:
268             print(f"  🔀 Merging {commits_behind} remote commits with local priority...")
269             result = run_command(
270                 ["git", "merge", f"origin/{branch}", "-X", "ours", "-m", f"Merge remote with local priority [{timestamp}]"],
271                 cwd=path,
272                 dry_run=dry_run,
273                 check=False
274             )
275             
276             if result.returncode != 0:
277                 if "Already up to date" in result.stdout or "Already up to date" in result.stderr:
278                     print(f"  ℹ️  Already up to date with remote")
279                 else:
280                     print(f"  ⚠️  Merge warning: {result.stderr}")
281                     # Continue to try push anyway
282             else:
283                 print(f"  ✅ Merged with local priority")
284         else:
285             print(f"  ℹ️  No remote commits to merge")
286     else:
287         print(f"  ℹ️  No remote branch yet, will create on push")
288     
289     # Step 5: Check if there's anything to push
290     if not is_ahead_of_remote(path, branch, dry_run):
291         print(f"  ℹ️  Nothing to push - already in sync with remote")
292         return True
293     
294     # Step 6: Push to remote
295     print(f"  📤 Pushing to remote...")
296     result = run_command(
297         ["git", "push", "origin", branch],
298         cwd=path,
299         dry_run=dry_run,
300         check=False
301     )
302     
303     if result.returncode != 0:
304         # Check if it's "no upstream branch" error
305         if "no upstream branch" in result.stderr.lower() or "has no upstream" in result.stderr.lower():
306             print(f"  📤 Setting upstream and pushing...")
307             result = run_command(
308                 ["git", "push", "-u", "origin", branch],
309                 cwd=path,
310                 dry_run=dry_run,
311                 check=False
312             )
313         
314         if result.returncode != 0:
315             # Check if it's "everything up-to-date"
316             if "everything up-to-date" in result.stdout.lower() or "everything up-to-date" in result.stderr.lower():
317                 print(f"  ✅ Already up to date with remote")
318                 return True
319             else:
320                 print(f"  ❌ Failed to push: {result.stderr}")
321                 return False
322         else:
323             print(f"  ✅ Successfully pushed to origin/{branch}")
324     else:
325         print(f"  ✅ Successfully pushed to origin/{branch}")
326     
327     return True

330 def output_data(data, fmt: str = "json"):
331     """Output data as JSON (default) or YAML to stdout."""
332     if fmt == "yaml":
333         import yaml
334         print(yaml.dump(data, default_flow_style=False, allow_unicode=True, sort_keys=False))
335     else:
336         print(json.dumps(data, separators=(",", ":"), ensure_ascii=False))

339 def main():
340     parser = argparse.ArgumentParser(
341         description="Git backup script - backs up paths to git remotes with local priority"
342     )
343     parser.add_argument(
344         "--message", "-m",
345         help="Custom commit message prefix",
346         default=None
347     )
348     parser.add_argument(
349         "--format", choices=["json", "yaml"], default="json",
350         help="Output format (default: json)"
351     )
352     parser.add_argument(
353         "--dry-run",
354         action="store_true",
355         help="Preview changes without executing"
356     )
357     parser.add_argument(
358         "--paths",
359         nargs="+",
360         help="Override configured backup paths",
361         default=None
362     )
363     
364     args = parser.parse_args()
365     
366     paths = args.paths if args.paths else BACKUP_PATHS
367     
368     if args.dry_run:
369         print("🔍 DRY RUN MODE - No changes will be made", file=sys.stderr)
370     
371     print(f"Backup paths configured: {len(paths)}", file=sys.stderr)
372     for p in paths:
373         print(f"  - {p}", file=sys.stderr)
374     
375     results = []
376     
377     for path in paths:
378         success = backup_path(path, args.message, args.dry_run)
379         results.append({"path": path, "success": success})
380     
381     success_count = sum(1 for r in results if r["success"])
382     fail_count = sum(1 for r in results if not r["success"])
383 
384     print(f"\nBackup complete: {success_count} succeeded, {fail_count} failed", file=sys.stderr)
385 
386     output_data({
387         "summary": {"successful": success_count, "failed": fail_count},
388         "dry_run": args.dry_run,
389         "results": results,
390     }, args.format)
391 
392     if fail_count > 0:
393         sys.exit(1)
