25 def run_command(cmd, cwd, dry_run=False, check=True):
26     """Run a shell command and return the result."""
27     if dry_run:
28         print(f"  [DRY-RUN] Would run: {' '.join(cmd)}")
29         return subprocess.CompletedProcess(cmd, 0, b"", b"")
30     
31     result = subprocess.run(
32         cmd,
33         cwd=cwd,
34         capture_output=True,
35         text=True
36     )
37     
38     if check and result.returncode != 0:
39         raise subprocess.CalledProcessError(
40             result.returncode, cmd, result.stdout, result.stderr
41         )
42     
43     return result
